import os
import random
def prepare_read(path, shuffle=False):
    images=[]
    labels=[]
    dirs = [path+d for d in os.listdir(path)]
    for i, dir_name in enumerate(dirs):
        print(dir_name)
        for fname in os.listdir(dir_name):
            images.append( dir_name + '/' + fname)
            labels.append(i)
        
    if (shuffle):
        temp = list(zip(images, labels))
        random.shuffle( temp )
        images,labels = zip(*temp) # unpack
        
    return images, labels

import cv2
def read_image(file_path, size=(224, 224)):
    img = cv2.imread(file_path, cv2.IMREAD_COLOR) #cv2.IMREAD_GRAYSCALE
    return cv2.resize(img, size, interpolation=cv2.INTER_CUBIC)

import numpy as np
CHANNELS = 3
def prep_data(images, size=(128, 128)):
    count = len(images)
    #data = np.ndarray((count, CHANNELS, ROWS, COLS), dtype=np.uint8) #通道在前
    data = np.ndarray((count, size[0], size[1], CHANNELS), dtype=np.uint8) #通道在後
    for i, image_file in enumerate(images):
        image = read_image(image_file, size)
        #data[i] = image.T # #通道在前
        data[i] = image #通道在後
        if i%250 == 0: print('Processed {} of {}'.format(i, count))
    print('shape:{}'.format(data.shape))
    return data

def read_data( path, shuffle=False, size=(224, 224)):
    all_image_path, labels = prepare_read(path, shuffle)
    images = prep_data( all_image_path, size=size )
    return images, labels
